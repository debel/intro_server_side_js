gulp - automate development tasks


        /* gulpFile.js */
        var gulp = require('gulp');
        
        gulp.task('lint-my-js', function () {
            //read js files
            return gulp.src('my/*.js')
                //pass them to jshint for static analysis
                .pipe(jshint({enforceall: true}))
                //jshint reports the results
                .pipe(jshint.reporter('default'));
        });
        

        /* gulpFile.js */
        gulp.task('build-js', function () {
            //get files from multiple locations
            return gulp.src(['array/of/paths', 'to/all/*.js'])
                //concatenate them into one file
                .pipe(concat('app.min.js'))
                //minify the concatenated file, remove whitespace, comments, etc
                .pipe(uglify())
                //save the result
                .pipe(gulp.dest('client/js/build/'));
        });

        //a task can be a series of other tasks 
        gulp.task('default', ['lint-js', 'build-js']);

running gulp from the console


        $ gulp
