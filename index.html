node.js - run javascript on the server

        
        /* server.js */
        var http = require('http'),
            //the server executes this callback
            //everytime someone connects to the server
            server = http.createServer(function (req, res) {
                res.end('Hello Web!');
        });
        
        //start the server on port 8080
        server.listen(8080);
  
        
        $ node server.js

gulp - automate development tasks


        /* gulpFile.js */
        var gulp = require('gulp');
        
        gulp.task('lint-my-js', function () {
            return gulp.src('my/*.js')  //take all js files
                .pipe(jshint({enforceall: true})) //pass them to jshint for static analysis
                .pipe(jshint.reporter('default')); //jshint reports the results
        });
        

        /* gulpFile.js */
        gulp.task('build-js', function () {
            return gulp.src(['array/of/paths', 'to/all/*.js']) //get files from multiple locations
                .pipe(concat('app.min.js')) //concatenate them into one file
                .pipe(uglify()) //minify the concatenated file, remove whitespace, comments, etc
                .pipe(gulp.dest('client/js/build/')); //save the result
        });

        gulp.task('default', ['lint-js', 'build-js']); //a task can be a series of other tasks
  

        $ gulp
