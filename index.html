gulp - automate development tasks


        /* gulpFile.js */
        var gulp = require('gulp');

        //define a task to check our code for errors
        gulp.task('lint-my-js', function () {
            /* read javascript files
            pass to jshint for static analysis
            jshint reports the results */
            return gulp.src('my/*.js')
                .pipe(jshint({enforceall: true}))
                .pipe(jshint.reporter('default'));
        });
        

        /* gulpFile.js */
        var concat = require('gulp-concat'),
            uglify = require('gulp-uglify');

        gulp.task('build-js', function () {
            /* get files from multiple locations
            concatenate and minify them into one file
            save the result to a specified location */
            return gulp.src(['array/of/paths', 'to/all/*.js'])
                .pipe(concat('app.min.js'))
                .pipe(uglify())
                .pipe(gulp.dest('client/js/build/'));
        });

        //a task can be a series of other tasks
        gulp.task('default', ['lint-js', 'build-js']);
        

            /* gulpFile.js */
            gulp.task('watch', function () {
                /* execute the default task every time
                one of these files changes */
                gulp.watch(['paths/to/project/files'], ['default']);
            });

running gulp from the console

            #execute a task by name
            $ gulp watch

socket.io: websockets

socket.io: server

        
        /* server.js */
        var io = require('socket.io')(http);

        io.on('connection', function(socket){
            console.log('a user connected');

            socket.on('new message', function(msg) {
                io.emit('new message', msg);
            }

            socket.on('disconnect', function(){
                console.log('user disconnected');
            });
        });

socket.io: client


            //we must add the socket.io client library path
